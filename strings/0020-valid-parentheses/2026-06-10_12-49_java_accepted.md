## Solution

```java
class Solution {
    public boolean isValid(String s) {
        Stack<Character> stack = new Stack<>();

        char[] arr = s.toCharArray();
        int i = 0;
        while(i!=s.length()){
            if(arr[i]=='(' || arr[i] == '{' || arr[i] == '[')
                stack.push(arr[i]);
            else{

                if(stack.isEmpty()) return false;
                
                if((stack.peek() == '(' && arr[i] == ')') || (arr[i] == '}' && stack.peek() == '{') ||( arr[i] == ']' && stack.peek()=='['))
                    stack.pop();
                else{
                    return false;
                }
            }
            i++;
        }
        if(stack.isEmpty()) return true;
        else    return false;
    }
}
```
